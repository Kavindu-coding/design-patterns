1. git init 
	- initialize git and create master branch

2. git status
	- Current status of the file structure

3. git add [file_name]
	- add a required file

4. git add --all or git add .
	- add all the files

5. git commit -m "[commit message]"
	- track the changes

6. git log 
	- log of the commits 

7. git remote add origin [git repo url]
	- add the existing folder to the repo

8. git branch -M main
	- creates a branch called 'main'

9. git push -u origin [branch name]
	- push the current commits or changes to the repositary 

10. git branch
	- list down available branches

11. git branch [new branch name]
	- creates a new branch

12. git checkout [new branch name]
	- switching to the current branch
